EXEMPLE 3 : calculator
#include <stdio.h>

int main() {

    float X, Y, Result, Op ;
    printf("input X value : ");
    scanf("%f", &X);
    printf("input Y value : ");
    scanf("%f", &Y);
    printf("1) addition    2) subtraction \n3) multiplication    4) division \n");
    printf("please choose Operation : ");
    scanf("%f", &Op);

    if( Op == 1 ) {
        Result = X + Y;
    }
    else if( Op == 2 ) {
        Result = X - Y;
    }
    else if( Op == 3 ) {
        Result = X * Y;
    }
    else if( Op == 4 ) {
        Result = X / Y;
    }
    else {
        printf("Error!!,Please select a number from 1 to 4");
    }
    printf("%f", Result);

return 0;
}
